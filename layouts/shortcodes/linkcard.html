{{- $url := .Get "url" -}} {{- if not $url -}} {{- errorf "linkcard shortcode requires a url parameter: %s" .Position -}} {{- end -}} {{- $title := .Get "title" | default $url -}} {{- $description := .Get "description" -}} {{- $image := .Get "image" -}} {{- $host := replaceRE "^https?://([^/]+).*" "$1" $url -}} {{- $site := .Get "site" | default $host -}} {{ $title }} {{- with $description }} {{ . }} {{- end }} {{ $site }} {{- if $image }} {{- else }} {{ substr $site 0 1 | upper }} {{- end }}
